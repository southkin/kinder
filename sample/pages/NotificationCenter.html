NotificationCenter
특정 엘리먼트의 이벤트가 아닌 전역 이벤트를 발생시켜 리스너 등록한 모든 작업을 수행하도록 하는 기능
NotificationCenter
여러 오브젝트를 생성할 필요가 없으므로 new 로 새 오브젝트를 생성하지 않고 NotificationCenter 자체를 오브젝트로 사용함
.addListener(notiName,notiId,fnc)
#arg
notiName - 이벤트 명
notiId - 이벤트리스너 아이디
fnc - 이벤트 발생시 수행할 함수
return : 없음
사용 예
NotificationCenter.addListener("event1", "notiSample", function(info){$(".cache").html(info.cache)});
.removeListener(notiName,notiId)
#arg
notiName - 이벤트 명
notiId - 이벤트리스너 아이디
return : 없음
사용 예
NotificationCenter.removeListener("event1", "notiSample");
.post(notiName,info)
#arg
notiName - 이벤트 명
info - 이벤트 인자
return : 없음
사용 예
NotificationCenter.post("event1",{sampleText:"sample"});
